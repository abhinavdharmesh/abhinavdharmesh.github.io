1. Interfacing an LED

An LED can be controlled using a digital pin.


int led = 13;

void setup() {
    pinMode(led, OUTPUT);
}

void loop() {
    digitalWrite(led, HIGH);
    delay(1000);
    digitalWrite(led, LOW);
    delay(1000);
}

2. Interfacing a Seven Segment Display

A seven-segment display can be controlled using multiple digital pins.


int segA = 2, segB = 3, segC = 4, segD = 5, segE = 6, segF = 7, segG = 8;

void setup() {
    for (int i = 2; i <= 8; i++) {
        pinMode(i, OUTPUT);
    }
}

void loop() {
    int zero[7] = {1,1,1,1,1,1,0}; 
    for (int i = 0; i < 7; i++) {
        digitalWrite(2 + i, zero[i]);
    }
    delay(1000);
}

3. Interfacing a Push Button

A button press toggles an LED.


int button = 2;
int led = 13;

void setup() {
    pinMode(button, INPUT);
    pinMode(led, OUTPUT);
}

void loop() {
    int state = digitalRead(button);
    digitalWrite(led, state);
}

5. Interfacing a DC Motor

A DC motor is controlled using PWM.


int motor = 9;

void setup() {
    pinMode(motor, OUTPUT);
}

void loop() {
    analogWrite(motor, 255);
    delay(2000);
    analogWrite(motor, 0);
    delay(2000);
}

6. Interfacing a Servo Motor

A servo motor rotates to a specific angle.


#include <Servo.h>

Servo myServo;

void setup() {
    myServo.attach(9);
}

void loop() {
    myServo.write(90);
    delay(1000);
    myServo.write(0);
    delay(1000);
}

8. Interfacing an Ultrasonic Sensor

Measures distance using sound waves.


int trig = 9, echo = 10;

void setup() {
    Serial.begin(9600);
    pinMode(trig, OUTPUT);
    pinMode(echo, INPUT);
}

void loop() {
    digitalWrite(trig, LOW);
    delayMicroseconds(2);
    digitalWrite(trig, HIGH);
    delayMicroseconds(10);
    digitalWrite(trig, LOW);

    long duration = pulseIn(echo, HIGH);
    int distance = duration * 0.034 / 2;

    Serial.print("Distance: ");
    Serial.print(distance);
    Serial.println(" cm");

    delay(1000);
}

9. Interfacing an IR Sensor

Detects an object using infrared light.


int ir = 2, led = 13;

void setup() {
    pinMode(ir, INPUT);
    pinMode(led, OUTPUT);
}

void loop() {
    int state = digitalRead(ir);
    digitalWrite(led, state == LOW ? HIGH : LOW);
}

10. Interfacing a Bluetooth Module

Wireless communication using HC-05.


#include <SoftwareSerial.h>

SoftwareSerial BT(10, 11);

void setup() {
    Serial.begin(9600);
    BT.begin(9600);
}

void loop() {
    if (BT.available()) {
        char data = BT.read();
        Serial.print("Received: ");
        Serial.println(data);
    }
}
